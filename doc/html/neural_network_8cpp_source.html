142 void NeuralNetwork::simpleBackProp(vector<double> correctOutput, double scale) {
143  for (int neuronIdx = outputs.size() - 1; neuronIdx >= 0; neuronIdx--) {
144  outputs[neuronIdx]->delta =
145  Neuron::sigmoidPrime(outputs[neuronIdx]->getNetInput())
146  * (correctOutput[neuronIdx]
147  - outputs[neuronIdx]->getActivity());
148  }
149  for (int neuronIdx = hidden.size() - 1; neuronIdx >= 0; neuronIdx--) {
150  double tempSum = 0;
151  for (SynapsePtr curOut : hidden[neuronIdx]->getChilds()) {
152  tempSum += curOut->weight * curOut->getOut()->delta;
153  }
154  hidden[neuronIdx]->delta = Neuron::sigmoidPrime(
155  hidden[neuronIdx]->getNetInput()) * tempSum;
156  }
157  for (SynapsePtr cur : synapses) {
158  cur->weightChange += scale * Synapse::learningRate
159  * cur->getIn()->getActivity() * cur->getOut()->delta;
160  }
161 }

163 vector<double> NeuralNetwork::backProp(vector<vector<double> > inputs,
164  vector<vector<double> > correctOutputs, vector<double> scaling) {
165  vector<double> errorSum;
166  for (SynapsePtr cur : synapses)
167  cur->weightChange = 0.;
168 #ifdef DEBUG
169  assert(inputs.size() == correctOutputs.size());
170  assert(inputs.size() == scaling.size());
171 #endif
172  for (unsigned i = 0; i < inputs.size(); ++i) {
173  feedForward(inputs[i]);
174  errorSum.push_back(calcErrorSum(correctOutputs[i], getOutput()));
175  simpleBackProp(correctOutputs[i], scaling[i]);
176  }
177  for (SynapsePtr cur : synapses) {
178  cur->weight += cur->weightChange;
179  cur->weightChange = 0.;
180  }
181  return errorSum;
182 }

205 void NeuralNetwork::saveNetwork(string s) const {
206  /* file format:
207  * 1. line: inputs, hiddens, outputs (numbering goes from 1 to (inputs+hiddens+outputs), 0 is bias)
208  * following lines: matrix representation of Synapse weights
209  *
210  * format of floating numbers: (+-)x.(DECIMALS)e(+-)xx, in total always 7+DECIMALS chars (+- is only one of them)
211  */
212  ofstream os(s);
213  os << inputs.size() << " ; " << hidden.size() << " ; " << outputs.size()
214  << endl;
215  os << setprecision(DECIMALS) << fixed << scientific;
216  for (unsigned row = 0;
217  row <= inputs.size() + hidden.size() + outputs.size(); ++row) {
218  NeuronPtr start = getNeuron(row);
219  for (unsigned col = 0;
220  col <= inputs.size() + hidden.size() + outputs.size(); ++col) {
221  bool found = false;
222  for (unsigned i = 0; i < start->getChilds().size(); ++i) {
223  if (start->getChilds()[i]->getOut()->getID() == col) {
224  found = true;
225  if (synapses[start->getChilds()[i]->getID()]->weight > 0)
226  os << "+";
227  os << synapses[start->getChilds()[i]->getID()]->weight;
228  }
229  }
230  if (!found)
231  os << string(DECIMALS + 7, ' ');
232  if (col < inputs.size() + hidden.size() + outputs.size())
233  os << " ; ";
234  else
235  os << endl;
236  }
237  }
238  os.close();
239 }
